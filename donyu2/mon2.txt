char name[20];

}data;

int main(){

data s[3];
for(i=0;i<3;i++){

scanf("%d",&s[i].eng);
printf("Math: ");
scanf("%d",&s[i].math);
printf("Jpn : ");

scanf("%d",&s[i].jpn);
	}
if((fp =fopen("mon2.txt","w"))==NULL){

}
